    Gibberish Audio Driver

  • ¶

    The audio driver has two tasks:

    1. Control the time by calling resume on the VM
    2. Create and play instruments
    /**
 * Create a VM plugin to add the audio driver
 * @return {Function} a VM plugin
 */
export default function init (vm, Gibberish, options) {
  const audio = new AudioDriver(Gibberish, options)
  createInstruments(audio.instruments, Gibberish)
  vm.addCommands(createCommands(audio))
  vm.audio = audio
  audio.start(vm)
  return audio
}

class AudioDriver {
  constructor (Gibberish, { bpm = 120 } = {}) {
    if (!Gibberish.context) Gibberish.init()
    this.Gibberish = Gibberish
    this.bpm = bpm
    this.sampleRate = Gibberish.context.sampleRate
    this.instruments = {}
    this.commands = createCommands(this)
  }

    Name Description Example
    @get-bpm Get the global tempo value "@pick", [1.25, 1.5, 0.75], "@get-bpm", "@*", "@set-bpm"
    @set-bpm Change the global tempo 120, "@set-bpm"
    @play Trigger a note. It uses the context to select the appropiate voice and parameters "@play" 
    const createCommands = audio => ({
  "@play": ({ context, error }) => {
    const instName = context.get("voice")
    const instrument = audio.instruments[instName]
    if (instrument) {
      instrument(context)
    } else {
      error(ERR_INST_MISSING(instName))
    }
  },
  "@set-bpm": ({ stack }) => {
    const bpm = parseFloat(stack.pop(), 10)
    if (bpm > 0) audio.bpm = bpm
  },
  "@scale-tempo": ({ stack }) => {
    const factor = parseFloat(stack.pop(), 10)
    if (factor) audio.bpm = audio.bpm * factor
  }
})
